/**
 * Backtracking (на русский язык переводится как «алгоритм с возвратом» или «поиск с возвратом») — это метод решения задач путем автоматического перебора всех возможных вариантов в поисках правильного ответа.
 * Если говорить просто, это метод проб и ошибок, работающий по принципу лабиринта.
 * 
 * Главный принцип работы:
 *
 * Вы идете по дорожке лабиринта (делаете выбор). Если вы упираетесь в тупик (текущий вариант не ведет к решению), вы делаете шаг назад (backtrack) к развилке и пробуете другую дорожку.
 * Этот алгоритм почти всегда реализуется через рекурсию.
 */

// Классический шаблон кода (как это выглядит в Java)
// В коде backtracking всегда имеет общую структуру:

void backtrack(Результат, Текущий_Путь, Данные) {
    // 1. Базовый случай (условие выхода)
    if (нашли_правильное_решение) {
        добавить_в_результат(Текущий_Путь);
        return;
    }

    // 2. Перебор всех возможных вариантов на данном шаге
    for (Вариант выбор : все_варианты) {
        if (выбор_допустим) {
            
            Текущий_Путь.add(выбор); // Сделать шаг (выбрать путь)
            
            backtrack(Результат, Текущий_Путь, Данные); // Уйти глубже в рекурсию
            
            Текущий_Путь.removeLast(); // ОТКАТ НАЗАД (BACKTRACK) — убираем выбор, чтобы попробовать следующий
        }
    }
}
