1. Переменные: изменяемые и неизменяемые

Для автоматического вывода типов используется ключевое слово var (как в Kotlin/Scala).

- final var или final Тип — константа. Значение нельзя изменить после создания.
- var или Тип — обычная переменная. Значение можно менять.

final var name = "Иван"; // Константа. Перезаписать name = "Петр" нельзя.
var age = 25;            // Переменная. Можно сделать age = 26.

3. Switch-выражения с Pattern Matching (аналог match / when)

// В Java 21 switch умеет проверять типы данных и возвращать значение с помощью оператора yield или стрелок ->.
int day = 1;
String dayName = switch (day) {
    case 1 -> "Monday";
    case 2 -> "Tuesday";
    default -> "Unknown"; // default обязателен
};

11. Ключевое слово static

// Ключевое слово static используется для создания членов класса (переменных и методов), которые принадлежат самому классу, а не его конкретным экземплярам (объектам).
// Когда используется static:
// 1 Общая память (Глобальные данные)
// 2 Утилитарные методы
// 3 Константы 
class Car {
    // Статическое поле — ОДНО на все объекты класса Car
    public static int carCount = 0; 

    // Обычные поля — у каждой машины свои brand и speed
    public final String brand;
    public int speed;

    public Car(String brand, int speed) {
        this.brand = brand;
        this.speed = speed;
        carCount++; // При создании каждой машины увеличиваем общий счетчик
    }
    // Статический метод — вызывается через имя класса Car.convertKmhToMph(...)
    // Ему не нужно знать бренд или скорость конкретной машины
    public static double convertKmhToMph(double kmh) {
        return kmh * 0.621371;
    }
}
// Использование в коде:
Car car1 = new Car("Toyota", 90);
Car car2 = new Car("BMW", 120);
// Обращаемся к статическому полю напрямую через класс
System.out.println(Car.carCount); // Выведет: 2 (всего создано две машины)
// Вызываем статический метод без создания объекта
double mph = Car.convertKmhToMph(100); 

12. Interfaces (Интерфейсы)

// Интерфейсы в Java могут содержать как абстрактные методы, так и методы с дефолтной реализацией (через ключевое слово default).
interface Animal {
    String sound();
    
    default String sleep() { 
        return "Zzz"; 
    }
}
// Реализация через ключевое слово implements
class Dog implements Animal {
    @Override
    public String sound() {
        return "Woof";
    }
}

13. Шаблоны строк (String Templates)

// В Java 21 появились шаблоны строк через префикс STR.. Переменные или выражения пишутся внутри блоков \{ }.
String name = "Alex";
// Больше не нужно складывать строки через плюсы
String greeting = STR."Hello, \{name}!"; 
System.out.println(greeting); // Выведет: Hello, Alex!
